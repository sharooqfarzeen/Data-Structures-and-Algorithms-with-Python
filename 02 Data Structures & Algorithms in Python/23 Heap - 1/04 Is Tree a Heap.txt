from os import *
from sys import *
from collections import *
from math import *

is_heap= True 
def is_leaf(node)->bool:
    if not node.left and not node.right:
        return True 
    return False 


def check(node) -> int :
    global is_heap
    if is_leaf(node):
        return node.data

    if not node.left:
        is_heap=False
        return node.data

    if node.left and not node.right and not is_leaf(node.left):
        is_heap=False 
        return node.data 
    max_left = check(node.left)
    max_right = node.data 
    if node.right:
        max_right=check(node.right)
    maxim = max(max_left,max_right ,node.data)
    if maxim==node.data:
        return node.data
    else:
        is_heap = False 
        return maxim
    

def isBinaryHeapTree(root):
    global is_heap 
    is_heap=True 
    check (root)
    return is_heap
